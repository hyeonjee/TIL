## forms.py를 사용하여 카테고리 만들기

### `forms.py`
```python
from django import forms
from .models import Product

POST_CATEGORY_CHOICES = [
        ('pub', 'pub'),
        ('booth', 'booth'),
        ('performance', 'performance'),
        ('etc', 'etc'),
    ]

class PostForm(forms.ModelForm):
    class Meta:
        model = Product
        fields = ('product_name', 'product_info', 'price', 'stock', 
        'image', 'category')
        labels = {
            'image: 사진'
            'product_name: 상품명',
            'product_info: 상품 소개',
            'price: 상품 가격',
            'stock: 재고',
            'category: 카테고리',
        }
        widgets = {
            'product_name': forms.TextInput(attrs={
                'class': 'form-group'
            }),
            'product_info': forms.Textarea(attrs={
                'class': 'form-group'
            }),
            'price': forms.TextInput(attrs={
                'class': 'form-group'
            }),
            'stock': forms.TextInput(attrs={
                'class': 'form-group'
            }),
            'category': forms.Select(choices=POST_CATEGORY_CHOICES)
        }

```  

`views.py`
```python
from .forms import PostForm #추가
```
